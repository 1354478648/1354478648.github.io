步骤五：创建一个饼图，用于显示男女比例¶

In [5]:
males = (df['Sex'] == 'male').sum()
females = (df['Sex'] == 'female').sum()

proportions = [males, females]

plt.figure(figsize=(6, 4))

plt.pie(
    # 数据
    proportions,
    
    # 设置每块的标签
    labels = ['男性', '女性'],
    
    # 设置是否添加阴影，默认为 False
    shadow = False,
    
    # 设置每块的颜色
    colors = ['blue','red'],
    
    # 设置每块的偏移量，用于突出显示某一块饼
    explode = (0.15 , 0),
    
    # 设置饼状图的起始角度
    startangle = 90,
    
    # 设置数值的精度和格式，例如'%1.2f%%'表示小数点后保留两位
    autopct = '%1.2f%%'
    )

# 显示为圆，避免压缩成椭圆
plt.axis('equal')
plt.title("男女比例")
# 简单地调整多图、单图标签，防止显示不全
plt.tight_layout()
plt.show()

    df['Age'],
    
    # Y轴的数据
    df['Fare'],
    
    # 设置点的颜色
    c = df['Sex'].map(colors),
    
    # 设置点的形状
    marker = 'o',
)

# 设置图表标题和轴标签

plt.ylabel('船票费用')
